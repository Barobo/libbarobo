%\phantomsection
\addcontentsline{toc}{section}{motionInchwormRight()}

\noindent
{\bf Synopsis}\\
\begin{verbatim}
#include <mobot.h>
int CMobot::motionInchwormRight();
\end{verbatim}

\noindent
{\bf Purpose}\\
Perform the inch-worm gait to the right.\\

\noindent
{\bf Return Value}\\
The function returns 0 on success and non-zero otherwise.\\

\noindent
{\bf Parameters}\\
None.\\

\noindent
{\bf Description}\\
This function causes the iMobot to perform a single cycle of the inchworm gait
to the right.\\
