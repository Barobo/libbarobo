%\phantomsection
\addcontentsline{toc}{section}{motionInchwormLeft()}

\noindent
{\bf Synopsis}\\
\begin{verbatim}
#include <imobot.h>
int CMobot::motionInchwormLeft();
\end{verbatim}

\noindent
{\bf Purpose}\\
Perform the inch-worm gait to the left.\\

\noindent
{\bf Return Value}\\
The function returns 0 on success and non-zero otherwise.\\

\noindent
{\bf Parameters}\\
None.\\

\noindent
{\bf Description}\\
This function causes the iMobot to perform a single cycle of the inchworm gait
to the left.\\
